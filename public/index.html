const res = await fetch('https://textkitapi.com/api/v1/summarize', {
  method: 'POST',
  headers: {
    'x-api-key': 'tk_live_your_key',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({ text }),
});
const { summary } = await res.json();
